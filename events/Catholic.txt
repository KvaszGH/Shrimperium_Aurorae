	desc = catholic_flavor.5.d
	picture = RELIGION_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		OR = { religion = catholic  religion = fraticelli }
		#NOT = { is_religion_enabled = protestant }
		NOT = { tag = PAP }
		NOT = { tag = FAP }
		#is_year = 1480
		ruler_age = 45
		NOT = {
			has_global_flag = pap_disables_indulgences_flag
		}
		is_excommunicated = no
		NOT = { has_country_modifier = papal_indulgence }
		NOT = {
			reverse_has_opinion_modifier = {
				who = PAP
				modifier = bought_indulgence
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	option = {
		name = catholic_flavor.5.a
		ai_chance = {
			factor = 10
			modifier = {
				factor = 0.5
				is_rival = PAP
			}
		}
		add_years_of_income = -1
		if = {
			limit = { NOT = { num_of_cities = 6 } }
			add_curia_treasur_small_effect = yes
		}
		else = {
			add_curia_treasury_medium_effect = yes
		}
		add_papal_influence = 10
		add_reform_desire = 0.002
	}
	option = {
		name = catholic_flavor.5.b
		ai_chance = {
			factor = 10
			modifier = {
				factor = 2
				has_global_flag = borgia_pope_global
			}
			modifier = {
				factor = 0.5
				alliance_with = PAP
			}
		}
		add_reform_desire = -0.002
		add_papal_influence = -10
		PAP = {
			add_opinion = {
				who = root
				modifier = catholic_obstructing_church
			}
		}
	}
}

# Curia controller based on dynasty
country_event = {
	id = catholic_flavor.11
	title = catholic_flavor.11.t
	desc = catholic_flavor.11.d
	picture = RELIGION_eventPicture
	
	is_triggered_only = yes
	hidden = yes
	
	trigger = {
		OR = {
			AND = {
				religion = catholic
				tag = PAP
			}
			AND = {
				religion = fraticelli
				tag = FAP
			}
		}
		OR = {
			AND = {
				ruler_is_foreigner = yes
				any_known_country = {
					religion = ROOT
					is_free_or_tributary_trigger = yes
					dynasty = ROOT
				}
			}
			AND = {
				is_subject = yes
				overlord = { religion = ROOT }
			}
			AND = {
				tag = PAP
				religion = catholic
				has_global_flag = papacy_permanent_curia_controller_flag
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			if = {
				limit = { is_subject = yes }
				random_country = {
					limit = {
						overlord_of = ROOT
						religion = ROOT
					}
					save_event_target_as = new_curia_controller
				}
			}
			random_country = {
				limit = {
					is_free_or_tributary_trigger = yes
					religion = ROOT
					dynasty = ROOT
				}
				save_event_target_as = new_curia_controller
			}
			every_known_country = {
				limit = {
					is_free_or_tributary_trigger = yes
					religion = ROOT
					dynasty = ROOT
					total_development = event_target:new_curia_controller
				}
				save_event_target_as = new_curia_controller
			}
			every_known_country = {
				limit = {
					is_free_or_tributary_trigger = yes
					religion = ROOT
					dynasty = ROOT
					overlord_of = ROOT
				}
				save_event_target_as = new_curia_controller
			}
			if = {
				limit = { 
					has_global_flag = papacy_permanent_curia_controller_flag 
					religion = catholic
					tag = PAP
				}
				PAP = {
					save_event_target_as = new_curia_controller
				}
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
		
	}
	
	option = {
		name = catholic_flavor.11.a
		
		transfer_papal_controller = event_target:new_curia_controller
	}
}
