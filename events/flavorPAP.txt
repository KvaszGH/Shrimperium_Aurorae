# The Counter-Reformation and the Reformation of the Catholic Faith
country_event = {
	id = flavor_pap.7
	title = "flavor_pap.7.t"
	desc = "flavor_pap.7.d"
	picture = REFORM_eventPicture
	
	fire_only_once = yes
	is_triggered_only = yes
	
	major = yes
	# this = country seeing popup, from = original event scope
	major_trigger = {
		religion_group = christian
	}
	
	trigger = {
		tag = PAP
		religion = catholic
		has_global_flag = catholic_antipopes_ended
		had_global_flag = {
			flag = catholic_antipopes_ended
			days = 3650
		}
		any_known_country = {
			religion = catholic
		}
		if = {
			limit = {
				has_dlc = "Emperor"
			}
			is_council_enabled = no
		}
		NOT = { has_global_flag = counter_reformation }
	}
	
	option = {
		name = "flavor_pap.EVTOPTA7"
		add_prestige = 10
		add_years_of_income = -0.5
		if = {
			limit = {
				corruption = 0.01
			}
			add_corruption = -1
		}
		if = {
			limit = {
				has_dlc = "Emperor"
			}
			custom_tooltip = flavor_pap.7.tt
			hidden_effect = {
				every_known_country = {
					limit = {
						religion = catholic
						any_owned_province = {
							has_cardinal = yes
						}
					}
					country_event = {
						id = flavor_pap.8
					}
				}
			}
		}
		set_global_flag = counter_reformation
		if = {
			limit = {
				has_dlc = "Emperor"
			}
			enable_council = yes
		}
	}
}

# Sublimis Deus
# by Daniel Tolman
country_event = {
	id = flavor_pap.9
	title = "flavor_pap.EVTNAME9"
	desc = "flavor_pap.EVTDESC9"
	picture = POPE_PREACHING_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		has_dlc = "Golden Century"
		religion = catholic
		culture_group = iberian
		capital_scope = {
			continent = europe
		}
		calc_true_if = {
			all_subject_country = {
				is_colonial_nation = yes
			}
			amount = 3
		}
		any_subject_country = {
			is_colonial_nation = yes
			num_of_cities = 15
		}
		exists = PAP
		NOT = { has_country_flag = had_sublimis_deus_event }
	}
	mean_time_to_happen = {
		months = 12
	}
	
	immediate = {
		hidden_effect = {
			set_country_flag = had_sublimis_deus_event
			every_country = {
				limit = {
					religion = catholic
					any_subject_country = {
						is_colonial_nation = yes
					}
					NOT = { has_country_flag = had_sublimis_deus_event }
				}
				country_event = { id = flavor_pap.10 days = 3 }
			}
		}
	}
	
	option = {
		name = "flavor_pap.9.a"
		add_papal_influence = 20
		PAP = {
			add_opinion = { who = ROOT modifier = opinion_sublimis_deus }
		}
		add_country_modifier = {
			name = "sublimis_deus"
			duration = 36500
		}
	}
	
	option = {
		name = "flavor_pap.9.b"
		every_subject_country = {
			limit = {
				is_colonial_nation = yes
			}
			add_country_modifier = {
				name = "sublimis_deus_not_enforced"
				duration = 36500
			}
		}
	}
}
