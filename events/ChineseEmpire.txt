			factor = 75
		}
		add_mandate_effect = yes
		add_meritocracy_large_effect = yes

				has_province_flag = resentment_to_sinicization

				name = "resistance_to_cultural_changes"

			}

		}

			}
		}
	}

country_event = {
	id = celestial_empire_events.4
	title = celestial_empire_events.4.t
	desc = celestial_empire_events.4.d
	picture = MERITOCRACY_eventPicture
	
	is_triggered_only = yes
	major = yes
	
	trigger = {
		is_emperor_of_china = yes
		#OR = { 
		#	ai = yes
		#}
		NOT = { has_disaster = bonfire_of_the_vanities_disaster }
		OR = { NOT = { imperial_mandate = 1 } is_subject = yes }
		any_known_country = {
			government = monarchy
			culture_group = east_asian
			is_subject = no
		}
	}
	
	immediate = {
		hidden_effect = {
			random_country = {
				limit = {
					government = monarchy
					culture_group = east_asian
					is_subject = no
					is_emperor_of_china = no
				}
				save_event_target_as = new_chinese_emperor
			}
			every_known_country = {
				limit = {
					government = monarchy
					culture_group = east_asian
					is_subject = no
					is_emperor_of_china = no
					total_development = event_target:new_chinese_emperor
				}
				save_event_target_as = new_chinese_emperor
			}
		}
	}
	
	option = {
		name = "celestial_empire_events.4.a"
		ai_chance = {
			factor = 75
		}
		hidden_effect = {
			add_government_reform = chinese_warlord
		}
		
		set_emperor_of_china = event_target:new_chinese_emperor
		
		hidden_effect = {
			event_target:new_chinese_emperor = {
				add_government_reform = celestial_empire
			}
			every_subject_country = {
				limit = { is_subject_of_type = tributary_state }
				event_target:new_chinese_emperor = {
					create_subject = {
						subject_type = tributary_state
						subject = PREV
					}
				}
			}
		}
	}
}
