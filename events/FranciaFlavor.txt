province_event = {
	id = ia_flavor_frw.2
	title = ia_flavor_frw.2.t
	desc = ia_flavor_frw.2.d
	picture = CITY_VIEW_eventPicture
	goto = ROOT
	hidden = yes
	
	trigger = {
		has_owner_religion = yes
		
		OR = {
			culture_group = french
			culture_group = germanic
			AND = {
				owner = { has_country_flag = frw_assimilate_iberia_flag }
				OR = {
					culture_group = iberian
					culture = visigothic
				}
			}
		}
		has_owner_accepted_culture = yes
		#NOT = { nationalism = 1 }
		
		owner = {
			primary_culture = frankish
		}
		any_country = { has_country_flag = frw_assimilation_enabled_flag }
	}
	
	mean_time_to_happen = {
		months = 240
	}
	
	option = {
		name = ia_flavor_frw.2.a
		
		if = {
			limit = { culture_group = french }
			change_culture = frankish
		}
		else_if = {
			limit = { culture_group = germanic }
			change_culture = frankish
		}
		else_if = {
			limit = { culture_group = iberian }
			change_culture = frankish
		}
		else_if = {
			limit = { culture = visigothic }
			change_culture = frankish
		}
	}
}

country_event = {
	id = ia_flavor_frw.5
	title = ia_flavor_frw.5.t
	desc = ia_flavor_frw.5.d
	picture = SIEGE_eventPicture

	is_triggered_only = yes
	
	option = {	# We got dismantled
		name = ia_flavor_frw.5.a
		
		release_all_subjects = yes
		white_peace = FRW
		hidden_effect = {
			every_owned_province = {
				limit = { 
					region = iberia_region 
					#culture = visigothic
				}
				add_devastation = 50
				add_base_production = -4
				add_base_manpower = -4
				add_base_tax = -4
				ROOT = {
					random_owned_province = {
						limit = { NOT = { region = iberia_region }  NOT = { culture = visigothic } }
						add_base_production = 2
						add_base_manpower = 2
						add_base_tax = 2
						change_culture = visigothic
					}
				}
			}
			every_owned_province = {
				limit = { 
					region = iberia_region 
					any_core_country = {
						exists = yes
						NOT = { tag = VAL }
					}
				}
				random_core_country = {
					limit = {
						exists = yes
						NOT = { tag = VAL }
					}
					PREV = {
						cede_province = PREV
					}
				}
			}
			every_owned_province = {
				limit = { 
					region = iberia_region 
					any_core_country = {
						exists = no
					}
				}
				random_core_country = {
					limit = {
						exists = no
					}
					ROOT = { release = PREV }
					change_religion = FROM
					#PREV = {
					#	cede_province = PREV
					#}
				}
			}
			every_owned_province = {
				limit = { 
					region = iberia_region 
				}
				add_core = SPA
				add_devastation = 50
			}
			if = {
				limit = { exists = SPA }
				every_owned_province = {
					limit = { 
						region = iberia_region 
						is_core = SPA
					}
					cede_province = SPA
				}
			}
			else = {
				release = SPA
				SPA = { change_religion = FROM }
			}
			FRW = {
				add_casus_belli = {
					target = SPA
					type = cb_restore_personal_union
					months = 1200
				}
			}
			if = {
				limit = { FRW = { ai = yes } }
				FRW = {
					declare_war_with_cb = {
						who = SPA
						casus_belli = cb_restore_personal_union
					}
				}
			}
		}
	}
}
