estate_industrialists_rheinholzfallergilde = {
	icon = privilege_dev_cost_forest
	max_absolutism = -10
	conditional_modifier = { trigger = { has_absolutism_reduction_for_estate_privileges = yes } modifier = { max_absolutism = 2 } }
	loyalty = 0.1
	influence = 0.1
	is_valid = {
		has_unlocked_estate_privilege = { estate_privilege = estate_industrialists_rheinholzfallergilde }
	}
	can_select = {
		any_owned_province = {
			OR = {
				has_terrain = forest
				has_terrain = woods
			}
		}
	}
	on_granted = {
		custom_tooltip = estate_burghers_forest_expansion_gt
		hidden_effect = {
			every_owned_province = {
				limit = {
					has_terrain = forest
				}
				add_province_modifier = {
					name = estate_industrialists_rheinholzfallergilde_forest
					duration = -1
				}
			}
			every_owned_province = {
				limit = {
					has_terrain = woods
				}
				add_province_modifier = {
					name = estate_industrialists_rheinholzfallergilde_woods
					duration = -1
				}
			}
		}
	}
	on_revoked = {
		custom_tooltip = estate_burghers_forest_expansion_rt
		hidden_effect = {
			every_owned_province = {
				limit = {
					OR = {
						has_province_modifier = estate_industrialists_rheinholzfallergilde_forest
						has_province_modifier = estate_industrialists_rheinholzfallergilde_woods
					}
				}
				remove_province_modifier = estate_industrialists_rheinholzfallergilde_forest
				remove_province_modifier = estate_industrialists_rheinholzfallergilde_woods
			}
		}
	}
	penalties = {
	}
	benefits = {
		development_cost_modifier = -0.05
	}
	ai_will_do = {
		factor = 100
	}
}
