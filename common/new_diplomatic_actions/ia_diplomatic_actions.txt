# Secure pilgrimage
secure_pilgrimage = {
	category = influence

	alert_index = 17
	alert_tooltip = secure_pilgrimage_alert_tt

	require_acceptance = yes # Whether the recipient gets an option to decline

	is_visible = {
		has_government_attribute = pilgrim_protector
		religion_group = christian
	}
	is_allowed = {
		has_government_attribute = pilgrim_protector
		religion_group = christian
		if = {
			limit = { FROM = { OR = { NOT = { religion_group = christian } religion = hellenic_pagan_reformed } } }
			custom_trigger_tooltip = {
				tooltip = pilgrim_protector_christian
				FROM = { AND = { religion_group = christian NOT = { religion = hellenic_pagan_reformed } } }
			}
		}
		if = {
			limit = { 
				OR = { 
					AND = { religion = catholic FROM = { tag = PAP } }
					AND = { religion = fraticelli FROM = { tag = FAP } }
				}
			}
			custom_trigger_tooltip = {
				tooltip = pilgrim_protector_pope
				FROM = { NOT = { OR = { tag = PAP tag = FAP } } }
			}
		}
		#if = {
		#	limit = { FROM = { is_subject_other_than_tributary_trigger = yes } }
		#	custom_trigger_tooltip = {
		#		tooltip = pilgrim_protector_subject
		#		FROM = { is_free_or_tributary_trigger = yes }
		#	}
		#}
		if = {
			limit = { NOT = { AND = { sailors = 1000 dip_power = 50 } } }
			custom_trigger_tooltip = {
				tooltip = pilgrim_protector_sailors
				sailors = 1000
				dip_power = 50
			}
		}
		if = {
			limit = { war_with = FROM }
			custom_trigger_tooltip = {
				tooltip = pilgrim_protector_war
				NOT = { war_with = FROM }
			}
		}
		if = {
			limit = { FROM = { capital_scope = { 	OR = { 
														region = egypt_region 
														region = arabia_region 
														region = mashriq_region 
													} } } }
			custom_trigger_tooltip = {
				tooltip = pilgrim_protector_holyland
				NOT = { FROM = { capital_scope = { 	OR = { 
														region = egypt_region 
														region = arabia_region 
														region = mashriq_region 
													} } } }
			}
		}
	}
	on_accept = {
		add_sailors = -1000
		add_dip_power = -50
		FROM = {
			add_stability = 1
			add_country_modifier = { 
				name = pilgrims_protected
				duration = 3650
			}
			capital_scope = { 
				add_trade_modifier = {
					who = ROOT
					duration = 3650
					power = 20
					key = pilgrim_influence
				}
			}
			country_event = { 
				id = ia_flavor_tem.1
			}
		}
		change_variable = {
			which = protected_pilgrims
			value = 1
		}
	}
	on_decline = {
	}

	ai_acceptance = { # This works similarly to variable_arithmetic_trigger, but does not support triggers generally. Instead, "limit" will be evaluated first and the rest ignored if it isn't passed.
		add_entry = {
			name = TRUST
			export_to_variable = {
				variable_name = ai_value
				value = trust
				who = FROM
				with = ROOT
			}			
			subtract_variable = { 
				which = ai_value
				value = 50
			}
			#1 trust = +4 ai points
			multiply_variable = {
				which = ai_value
				value = 0.4
			}
		}
		add_entry = {
			name = OPINION
			export_to_variable = {
				variable_name = ai_value
				value = opinion
				who = FROM
				with = ROOT
			}
			multiply_variable = {
				which = ai_value 
				value = 0.2
			}
		}
		add_entry = {
			name = DIPLO_IS_RIVAL
			limit = {
				FROM = {
					ai_attitude = {
						who = ROOT
						attitude = attitude_rivalry
					}
				}
			}
			export_to_variable = {
				variable_name = ai_value
				value = monthly_income
				who = ROOT
			}
			multiply_variable = {
				which = ai_value
				value = 0
			}
			change_variable = {
				which = ai_value
				value = -1000
			}
			
			#multiply_variable = {
			#	which = ai_value
			#	value = -1000
			#}
		}
		add_entry = {
			name = STABILITY_AI_MODIFIER
			limit = {
				FROM = {
					stability = 3
				}
			}
			export_to_variable = {
				variable_name = ai_value
				value = stability
				who = FROM
				with = ROOT
			}
			multiply_variable = {
				which = ai_value 
				value = -25
			}
		}
		add_entry = {
			name = diplomatic_reputation
			limit = { }
			export_to_variable = {
				variable_name = ai_value
				value = trigger_value:diplomatic_reputation
			}
			#1 rep = 2
			multiply_variable = {
				which = ai_value
				value = 2
			}
		}
		add_entry = {
			name = STABILITY_AI_MODIFIER
			export_to_variable = {
				variable_name = ai_value
				value = stability
				who = FROM
				with = ROOT
			}
			multiply_variable = {
				which = ai_value 
				value = -25
			}
		}
	}

	ai_will_do = { # this is a trigger
		has_opinion = {
			who = FROM
			value = 75
		}
		FROM = {
			NOT = { stability = 2 }
		}
		NOT = { is_rival = FROM }
		NOT = { is_enemy = FROM }
		#trust = {
		#	who = FROM
		#	value = 65
		#}
	}
}

		always = yes
	}

	require_acceptance = yes

	alert_index = 20
	alert_tooltip = ia_enforce_peace_cn_of_subject_alert_tooltip

	is_visible = {
		any_subject_country = {
			any_subject_country = { 
				is_colonial_nation = yes
				any_war_enemy_country = {
					NOT = { war_with = ROOT }
				}
			}
		}
		FROM = {
			NOT = { war_with = ROOT }
			any_war_enemy_country = {
				is_colonial_nation = yes
				overlord = { is_subject_of = ROOT }
			}
		}
	}
	is_allowed = {
		FROM = {
			NOT = { war_with = ROOT }
			any_war_enemy_country = {
				is_colonial_nation = yes
				overlord = { is_subject_of = ROOT }
			}
		}
	}
	on_accept = {
		FROM = {
			every_war_enemy_country = {
				limit = {
					is_colonial_nation = yes
					overlord = { is_subject_of = ROOT }
				}
				white_peace = FROM
			}
		}
	}
	on_decline = {
		FROM = {
			every_war_enemy_country = {
				limit = {
					is_colonial_nation = yes
					overlord = { is_subject_of = ROOT }
				}
				ROOT = { join_all_defensive_wars_of = PREV }
				ROOT = { join_all_offensive_wars_of = PREV }
			}
		}
	}

	ai_will_do = { # this is a trigger
		FROM = { war_score = 10 }
	}

	ai_acceptance = {
        add_entry = {
            name = DIPLO_OPPINION
			export_to_variable = {
				variable_name = ai_value
				value = opinion
				who = FROM
				with = ROOT
			}
			multiply_variable = {
				which = ai_value 
				value = 0.125
			}
        }
		add_entry = {
			name = AI_TRUST_TOWARDS_ROOT
			export_to_variable = {
				variable_name = ai_value
				value = trust
				who = FROM
				with = ROOT
			}			
			subtract_variable = { 
				which = ai_value
				value = 50
			}
			#1 trust = +4 ai points
			multiply_variable = {
				which = ai_value
				value = 0.5
			}
		}
		add_entry = {
			name = AI_DIPLOMATIC_REPUTATION
			export_to_variable = {
				variable_name = ai_value
				value = trigger_value:diplomatic_reputation
			}
			#1 rep = 3
			multiply_variable = {
				which = ai_value
				value = 3
			}
		}
		add_entry = {
			name = AI_TOTAL_DEVELOPMENT
			export_to_variable = {
				variable_name = ai_value
				value = total_development
				who = FROM
			}
			#4 dev = -1
			multiply_variable = {
				which = ai_value
				value = -0.25
			}
		}
	}
}

ia_request_excommunication_action = {
	category = papacy

	require_acceptance = no

	is_visible = {
		is_revolutionary = no
		OR = {
			religion = fraticelli
			religion = catholic
		}
		FROM = { religion = ROOT }
		FROM = {
			NOT = { tag = PAP }
			NOT = { tag = FAP }
		}
		if = {
			limit = { religion = catholic }
			NOT = { war_with = PAP }
		}
		if = {
			limit = { religion = fraticelli }
			NOT = { war_with = FAP }
		}
	}

	is_allowed = {
		is_excommunicated = no
		FROM = { religion = ROOT }
		FROM = { is_excommunicated = no }
		FROM = { is_papal_controller = no }
		FROM = { is_lesser_in_union = no }
		OR = {
			papal_influence = FROM
			has_reform = papacy_reform
		}
		OR = {
			papal_influence = 90
			AND = {
				is_papal_controller = yes
				papal_influence = 45
			}
			has_reform = papacy_reform
		}
	}

	on_accept = {
		if = {
			limit = { ai = no }
			add_papal_influence = -45
		}
		excommunicate = FROM
		reverse_add_opinion = {
			who = FROM
			modifier = excommunicated_us
		}
	}

	ai_will_do = {
		OR = {
			FROM = { OR = { is_rival = ROOT 	is_enemy = ROOT } }
			ai_attitude = {
				who = FROM
				attitude = attitude_hostile
			}
			ai_attitude = {
				who = FROM
				attitude = attitude_outraged
			}
			ai_attitude = {
				who = FROM
				attitude = attitude_disloyal
			}
			ai_attitude = {
				who = FROM
				attitude = attitude_rebellious
			}
			ai_attitude = {
				who = FROM
				attitude = attitude_rivalry
			}
			war_with = FROM
		}
	}
}

ia_subject_revolutionary_status = {
	category = influence

	require_acceptance = no

	is_visible = {
		NOT = { has_government_attribute = is_peasants_republic }
		FROM = { is_subject_of = ROOT }
		OR = {
			AND = {
				is_revolutionary = no
				FROM = { is_revolutionary = yes }
			}
			AND = {
				is_revolutionary = yes
				FROM = { is_revolutionary = no }
			}
		}
	}

	is_allowed = {
		FROM = { is_subject_of = ROOT }
		FROM = { NOT = { liberty_desire = 40 } }
		if = {
			limit = { is_revolutionary = yes }
			FROM = { is_revolutionary = no }
		}
		else = {
			FROM = { is_revolutionary = yes }
		}
	}

	on_accept = {
		if = {
			limit = { is_revolutionary = yes }
			FROM = { 
				add_liberty_desire = 25
				enact_revolution_effect = yes 
				every_owned_province = {
					limit = {
						NOT = { has_revolution_in_province = yes }
					}
					pretender_rebels = 2
				}
			}
		}
		else = {
			FROM = {
				add_liberty_desire = 25
				country_event = { id = center_of_revolution.1500 }
				every_owned_province = {
					limit = {
						has_revolution_in_province = yes
					}
					remove_revolution_from_province_effect = yes
					revolutionary_rebels = 2
				}
				if = {
					limit = { ROOT = { government = republic }}
					change_government = republic
				}
				else_if = {
					limit = { ROOT = { government = theocracy }}
					change_government = theocracy
				}
			}
		}
	}

	ai_will_do = {
		is_at_war = no
	}
}
