dismantle_nation_effect = {
	release_all_subjects = yes
	
	every_owned_province = {
		limit = {
			any_core_country = {
				primary_culture = PREV
				PREV = {
					NOT = { owned_by = PREV }
				}
			}
		}
		#remove_core = ROOT
		random_core_country = {
			limit = {
				PREV = {
					culture = PREV
					NOT = { PREV = { tag = ROOT } }
				}
			}
			if = {
				limit = { NOT = { exists = PREV } }
				ROOT = { release = PREV }
			}
			PREV = {
				cede_province = PREV
			}
		}
	}
	every_owned_province = {
		limit = {
			any_core_country = {
				culture_group = PREV
				PREV = {
					NOT = { owned_by = PREV }
				}
			}
		}
		#remove_core = ROOT
		random_core_country = {
			limit = {
				PREV = {
					culture_group = PREV
					NOT = { PREV = { tag = ROOT } }
				}
			}
			if = {
				limit = { NOT = { exists = PREV } }
				ROOT = { release = PREV }
			}
			PREV = {
				cede_province = PREV
			}
		}
	}
}

ia_BYZ_convert_to_greek_culture = { #It's here in case I want to add anatolian-themed cultures.
	if = {
		limit = {
			culture = armenian
			owner = {
				has_country_flag = tre_wont_convert_armenians
			}
		}
		change_culture = cilician
	}
	else_if = {
		limit = {
			OR = {
				area = erzurum_area
				area = kastamonu_area
			}
		}
		change_culture = pontic_greek
	}
	else_if = {
		limit = {
			area = karaman_area
		}
		change_culture = cappadocian_greek
	}
	else = {
		change_culture = anatolian
	}
}

rhe_add_prod_from_manu = {
	if = {
		limit = {
			FROM = { has_country_flag = rhe_enabled_dev_from_manus }
			NOT = { rhe_gained_dev_from_manu = { building = $building$ } }
		}
		add_base_$type$ = $value$
		hidden_effect = {
			set_province_flag = rhe_built_manu_dev_$building$
		}
	}
}

rhe_remove_prod_from_manu = {
	if = {
		limit = {
			FROM = { has_country_flag = rhe_enabled_dev_from_manus }
			rhe_gained_dev_from_manu = { building = $building$ }
		}
		add_base_$type$ = -$value$
		if = {
			limit = {
				NOT = { base_$type$ = 1 }
			}
			set_base_$type$ = 1
		}
		hidden_effect = {
			clr_province_flag = rhe_built_manu_dev_$building$
		}
	}
}
